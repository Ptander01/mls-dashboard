Priority 1 — Squad Depth Fix (Quick, High Impact)
File: resilienceUtils.ts
Switch from inverted HHI to Gini coefficient on top-15 players by minutes. The top-15 filter removes garbage-time minutes that compress variance. Gini is more spread-sensitive than HHI for this distribution shape.

export function squadDepthGini(teamId: string, players: Player[]): number {
  const squad = players
    .filter(p => p.team === teamId && p.minutes > 0)
    .sort((a, b) => b.minutes - a.minutes)
    .slice(0, 15)  // top 15 only
 
  const n = squad.length
  if (n === 0) return 0
 
  const total = squad.reduce((s, p) => s + p.minutes, 0)
  const sorted = [...squad].sort((a, b) => a.minutes - b.minutes)
 
  let numerator = 0
  sorted.forEach((p, i) => {
    numerator += (2 * (i + 1) - n - 1) * p.minutes
  })
 
  // Returns 0 (perfectly equal) to 1 (one player plays everything)
  // Invert: depthScore = 1 - gini, higher = deeper squad
  return numerator / (n * total)
}
