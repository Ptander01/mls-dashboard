6. Sample Code Snippet (Shot Map Component)

Here's a minimal example using React Three Fiber:

```tsx
import { useLoader } from '@react-three/fiber'
import { Sphere } from '@react-three/drei'
import * as THREE from 'three'

interface Shot {
  location: [number, number]
  outcome: string
  teamColor: string
}

function ShotMap({ shots }: { shots: Shot[] }) {
  return (
    <group>
      {shots.map((shot, i) => {
        const [x, y] = shot.location
        // Convert StatsBomb coords (0-120, 0-80) to Three.js (-60..60, -40..40)
        const posX = x - 60
        const posZ = y - 40 // assuming Y is up, we use XZ plane
        const color = shot.outcome === 'Goal' ? '#00ff00' :
                      shot.outcome === 'Saved' ? '#ffaa00' : '#ff0000'
        return (
          <Sphere
            key={i}
            position={[posX, 0.5, posZ]} // slightly above pitch
            args={[0.5, 16, 16]}
          >
            <meshStandardMaterial color={color} emissive={color} emissiveIntensity={0.5} />
          </Sphere>
        )
      })}
    </group>
  )
}
```
